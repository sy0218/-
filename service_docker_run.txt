# 컬렉터
docker run -d -it \
  --name collector \
  -v /etc/localtime:/etc/localtime:ro \
  -v /etc/timezone:/etc/timezone:ro \
  collector:v1 \
  /bin/bash

# 컨슈머
docker run -d -it \
  --name consumer \
  -v /etc/localtime:/etc/localtime:ro \
  -v /etc/timezone:/etc/timezone:ro \
  -v /nfs:/nfs \
  consumer:v1 \
  /bin/bash
